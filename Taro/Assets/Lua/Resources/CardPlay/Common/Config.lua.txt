Config = {}

-- 牌点
Config.Figure = {
    figure2 = 1,
    figure3 = 2,
    figure4 = 3,
    figure5 = 4,
    figure6 = 5,
    figure7 = 6,
    figure8 = 7,
    figure9 = 8,
    figure10 = 9,
    figureJ = 10,
    figureQ = 11,
    figureK = 12,
    figureA = 13,
}

-- 花色
Config.Pattern = { diamond = 1, club = 2, heart = 3, spade = 4 }

Config.Card = {
    FigureNum = function(self)
        return Config.Figure[self.figure]
    end,

    PatternNum = function(self)
        return Config.Pattern[self.pattern]
    end,

    -- 大小对比
    Compare = function(self, b)
        local aFigureNum = self:FigureNum()
        local bFigureNum = b:FigureNum()

        -- 先进行点数判断
        if aFigureNum > bFigureNum then
            return 1
        elseif aFigureNum < bFigureNum then
            return -1
        else
            -- 点数相同时，进行花色判断
            local aPatternNum = self:PatternNum()
            local bPatternNum = self:PatternNum()
            if aPatternNum > bPatternNum then
                return 1
            elseif aPatternNum < bPatternNum then
                return -1
            else
                return 0
            end
        end

    end
}

-- 创建新卡片
function Config.NewCard(_figure, _pattern)
    local o = setmetatable({ figure = _figure, pattern = _pattern }, { __index = Config.Card })
    return o
end

return Config
